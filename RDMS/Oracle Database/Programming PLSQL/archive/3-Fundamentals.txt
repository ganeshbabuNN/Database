Using Variables
================

Hello World Variables.sql
DECLARE
  v_message VARCHAR2(100);
  v_date DATE;
BEGIN
  DBMS_OUTPUT.put_line('Hello World!');
END;
/

Assigning the variable
======================
here just we are assigning the variable and start to use it
DECLARE
  v_message VARCHAR2(100);
  v_date DATE;
BEGIN
  v_message := 'Hello World' ;
  DBMS_OUTPUT.put_line('Hello World!');
END;
/

Using the variable
==================
DECLARE
  v_message VARCHAR2(100);
  v_date DATE;
BEGIN
  v_message := 'Hello World' ;
  DBMS_OUTPUT.put_line(v_message);
END;
/

Using Constants
================
-constant are the variable which does not change its value for life of the program
- the constant variable must be initialized
DECLARE
  v_message CONSTANT VARCHAR2(100):= 'Hello World' ;
  v_date DATE;
BEGIN
  DBMS_OUTPUT.put_line(v_message);
END;
/

Adding Funtinality of the Date
========================
--FM is a format modifiers
-- for upper case of a month, using MONTH as UPPER Case or lower case for the format mask that you are using

Hello World Final.sql
DECLARE
  v_message CONSTANT VARCHAR2(100) := 'Hello World!';
  v_date DATE := SYSDATE;
BEGIN
  DBMS_OUTPUT.put_line(
        v_message ||
        ' The date and time is ' ||
        to_char(v_date, 'FMDay') || ' on ' ||
        to_char(v_date, 'FMDD Month, YYYY') ||
        ' @ ' ||
        to_char(v_date, 'HH24:MI:SS')
        );
END;
/
